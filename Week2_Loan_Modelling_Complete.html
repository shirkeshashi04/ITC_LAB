3) Data Cleaning


Cleaning steps:
- Standardize column names
- Fix negative Experience values (set to median of non-negative experience)
- Remove ID (identifier)
- Treat ZIPCode as a categorical-like variable (do not scale it; keep only if you one-hot encode; here we drop it)
- Check missing values
In [16]:
# Standardize column names (your file uses ZIPCode)
df = df.rename(columns={"ZIP Code": "ZIPCode"})  # safe rename if the alternate name exists

# Fix negative Experience (common in this dataset)
if "Experience" in df.columns:
    neg_exp = (df["Experience"] < 0).sum()
    if neg_exp > 0:
        median_exp = df.loc[df["Experience"] >= 0, "Experience"].median()
        df.loc[df["Experience"] < 0, "Experience"] = median_exp
        print(f"Fixed {neg_exp} negative Experience values using median={median_exp}.")
    else:
        print("No negative Experience values found.")

# Drop obvious ID-like columns
if "ID" in df.columns:
    df = df.drop(columns=["ID"])

We train a regression model to predict CCAvg and evaluate using RMSE and R². We also interpret coefficients.

# 7) Prepare regression data
# Features: all except CCAvg and Personal_Loan
reg_features = [c for c in df.columns if c not in [target_reg, target_clf]]

X_reg = df[reg_features].copy()
y_reg = df[target_reg].copy()

# Train-test split
X_train_reg, X_test_reg, y_train_reg, y_test_reg = train_test_split(
    X_reg, y_reg, test_size=0.30, random_state=42
)

# Scaling (recommended for linear regression coefficient interpretability and stability)
scaler_reg = StandardScaler()
X_train_reg_sc = scaler_reg.fit_transform(X_train_reg)
X_test_reg_sc  = scaler_reg.transform(X_test_reg)

# Train model
lin_reg = LinearRegression()
lin_reg.fit(X_train_reg_sc, y_train_reg)

# Predictions
y_train_pred_reg = lin_reg.predict(X_train_reg_sc)
y_test_pred_reg  = lin_reg.predict(X_test_reg_sc)

# Metrics
#train_rmse = mean_squared_error(y_train_reg, y_train_pred_reg, squared=False)
train_rmse= np.sqrt(mean_squared_error(y_train_reg, y_train_pred_reg))
test_rmse= np.sqrt(mean_squared_error(y_test_reg, y_test_pred_reg))
#test_rmse  = mean_squared_error(y_test_reg,  y_test_pred_reg,  squared=False)
train_r2   = r2_score(y_train_reg, y_train_pred_reg)
test_r2    = r2_score(y_test_reg,  y_test_pred_reg)

print(f"Train RMSE: {train_rmse:.4f}")
print(f"Test RMSE : {test_rmse:.4f}")
print(f"Train R2  : {train_r2:.4f}")
print(f"Test R2   : {test_r2:.4f}")

12) Threshold Tuning (Optional)


Lowering the probability threshold usually increases recall but decreases precision.
This is useful for marketing use cases where missing a potential borrower can be more costly than contacting a non-borrower.
In [ ]:
from sklearn.metrics import precision_recall_curve, precision_score, recall_score, f1_score

y_test_proba = log_reg.predict_proba(X_test_clf_sc)[:, 1]

thresholds_to_try = [0.20, 0.30, 0.35, 0.40, 0.50, 0.60, 0.70]
print("=== Threshold Sweep (Test Set) ===")
for thr in thresholds_to_try:
    y_pred_thr = (y_test_proba >= thr).astype(int)
    p = precision_score(y_test_clf, y_pred_thr, zero_division=0)
    r = recall_score(y_test_clf, y_pred_thr, zero_division=0)
    f = f1_score(y_test_clf, y_pred_thr, zero_division=0)
    cm = confusion_matrix(y_test_clf, y_pred_thr)
    print(f"threshold={thr:.2f} | Precision={p:.3f} Recall={r:.3f} F1={f:.3f} | CM={cm.tolist()}")

prec, rec, thr = precision_recall_curve(y_test_clf, y_test_proba)

# XGBoost classification (with safe import)
try:
    from xgboost import XGBClassifier
    xgb_ok = True
except Exception as e:
    xgb_ok = False
    print("XGBoost not available:", e)

if xgb_ok:
    # Handle class imbalance using scale_pos_weight
    neg = (y_train_clf == 0).sum()
    pos = (y_train_clf == 1).sum()
    scale_pos_weight = neg / max(pos, 1)

    xgb = XGBClassifier(
        n_estimators=300,
        max_depth=4,
        learning_rate=0.05,
        subsample=0.9,
        colsample_bytree=0.9,
        reg_lambda=1.0,
        random_state=42,
        eval_metric="logloss",
        scale_pos_weight=scale_pos_weight
    )

    xgb.fit(X_train_clf_sc, y_train_clf)
    y_test_pred_xgb = xgb.predict(X_test_clf_sc)

    print("XGBOOST TEST METRICS")
    print(classification_report(y_test_clf, y_test_pred_xgb, zero_division=0))
    print("Confusion Matrix [[TN, FP],[FN, TP]]:")
    print(confusion_matrix(y_test_clf, y_test_pred_xgb))

    # Feature importance (based on the scaled feature set order)
    importances = xgb.feature_importances_
    idx = np.argsort(importances)

    plt.figure(figsize=(8, 8))
    plt.title("XGBoost Feature Importances")
    plt.barh(range(len(idx)), importances[idx])
    plt.yticks(range(len(idx)), [clf_features[i] for i in idx])
    plt.xlabel("Importance")
    plt.tight_layout()
    plt.show()

- Regression creates Predicted_CCAvg to represent expected spend behavior.
- Classification uses Predicted_CCAvg (not actual CCAvg) to predict loan acceptance.
- With imbalanced classes, prioritize metrics for class 1 (loan): precision/recall/F1.
- For marketing, higher recall can be acceptable if outreach cost is low.
